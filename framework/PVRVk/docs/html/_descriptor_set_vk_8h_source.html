776 // For smart pointer reference counting we need to keep alive the descriptor set binding entries to do this we place them in an array kept alive by the DescriptorSet itself
777 // This means that the caller application can let resources go out scope and the Descriptor set "keepAlive" array will keep them alive as long as needed
778 inline void WriteDescriptorSet::updateKeepAliveIntoDestinationDescriptorSet() const
779 {
780  // Get the keep alive entry for the current binding
781  auto& keepAlive = getDescriptorSet()->_keepAlive[this->_dstBinding];
782 
783  // Handle BufferInfo entries
784  if (_infoType == InfoType::BufferInfo)
785  {
786  for (uint32_t i = 0; i < _infos.size(); ++i)
787  {
788  // Ensure the into entry is valid
789  if (_infos[i].isValid()) { keepAlive[i] = _infos[i].bufferInfo.buffer; }
790  }
791  }
792  // Handle ImageInfo entries
793  else if (_infoType == InfoType::ImageInfo)
794  {
