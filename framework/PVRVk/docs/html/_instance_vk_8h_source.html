306  DebugUtilsMessageSeverityFlagsEXT inMessageSeverity, DebugUtilsMessageTypeFlagsEXT inMessageTypes, const DebugUtilsMessengerCallbackData& inCallbackData)
307  {
308  VkDebugUtilsMessengerCallbackDataEXT vkCallbackData = {};
309  vkCallbackData.sType = static_cast<VkStructureType>(StructureType::e_DEBUG_UTILS_MESSENGER_CALLBACK_DATA_EXT);
310  vkCallbackData.flags = static_cast<VkDebugUtilsMessengerCallbackDataFlagsEXT>(inCallbackData.getFlags());
311  vkCallbackData.pMessageIdName = inCallbackData.getMessageIdName().c_str();
312  vkCallbackData.messageIdNumber = inCallbackData.getMessageIdNumber();
313  vkCallbackData.pMessage = inCallbackData.getMessage().c_str();
314 
315  pvrvk::ArrayOrVector<VkDebugUtilsLabelEXT, 4> vkQueueLabels(inCallbackData.getNumQueueLabels());
316  pvrvk::ArrayOrVector<VkDebugUtilsLabelEXT, 4> vkCmdBufLabels(inCallbackData.getNumCmdBufLabels());
317  pvrvk::ArrayOrVector<VkDebugUtilsObjectNameInfoEXT, 4> vkObjectNames(inCallbackData.getNumObjects());
318 
319  // Add queue labels
320  if (inCallbackData.getNumQueueLabels())
321  {
322  vkCallbackData.queueLabelCount = inCallbackData.getNumQueueLabels();
323  for (uint32_t i = 0; i < inCallbackData.getNumQueueLabels(); ++i)
324  {
325  DebugUtilsLabel queueLabel = inCallbackData.getQueueLabel(i);
326 
327  vkQueueLabels[i].sType = static_cast<VkStructureType>(StructureType::e_DEBUG_UTILS_LABEL_EXT);
328  vkQueueLabels[i].pLabelName = queueLabel.getLabelName().c_str();
329  vkQueueLabels[i].color[0] = queueLabel.getR();
330  vkQueueLabels[i].color[1] = queueLabel.getG();
331  vkQueueLabels[i].color[2] = queueLabel.getB();
332  vkQueueLabels[i].color[3] = queueLabel.getA();
333  }
334 
335  vkCallbackData.pQueueLabels = vkQueueLabels.get();
336  }
337 
338  // Add command buffer labels
339  if (inCallbackData.getNumCmdBufLabels())
340  {
341  vkCallbackData.cmdBufLabelCount = inCallbackData.getNumCmdBufLabels();
342  for (uint32_t i = 0; i < inCallbackData.getNumCmdBufLabels(); ++i)
343  {
344  DebugUtilsLabel cmdBufLabel = inCallbackData.getCmdBufLabel(i);
345 
346  vkCmdBufLabels[i].sType = static_cast<VkStructureType>(StructureType::e_DEBUG_UTILS_LABEL_EXT);
347  vkCmdBufLabels[i].pLabelName = cmdBufLabel.getLabelName().c_str();
348  vkCmdBufLabels[i].color[0] = cmdBufLabel.getR();
349  vkCmdBufLabels[i].color[1] = cmdBufLabel.getG();
350  vkCmdBufLabels[i].color[2] = cmdBufLabel.getB();
351  vkCmdBufLabels[i].color[3] = cmdBufLabel.getA();
352  }
353 
354  vkCallbackData.pCmdBufLabels = vkCmdBufLabels.get();
355  }
356 
357  // Add object names
358  if (inCallbackData.getNumObjects())
359  {
360  vkCallbackData.objectCount = inCallbackData.getNumObjects();
361  for (uint32_t i = 0; i < inCallbackData.getNumObjects(); ++i)
362  {
363  DebugUtilsObjectNameInfo objectName = inCallbackData.getObject(i);
364 
365  vkObjectNames[i].sType = static_cast<VkStructureType>(StructureType::e_DEBUG_UTILS_OBJECT_NAME_INFO_EXT);
366  vkObjectNames[i].pObjectName = objectName.getObjectName().c_str();
367  vkObjectNames[i].objectType = static_cast<VkObjectType>(objectName.getObjectType());
368  vkObjectNames[i].objectHandle = objectName.getObjectHandle();
369  }
370 
371  vkCallbackData.pObjects = vkObjectNames.get();
372  }
