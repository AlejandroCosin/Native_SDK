247  void calcDynamicAlignment(BufferUsageFlags usage = BufferUsageFlags::UniformBuffer, const uint64_t minUboDynamicAlignment = 0, const uint64_t minSsboDynamicAlignment = 0)
248  {
249  _minDynamicAlignment = 0;
250 
251  uint64_t uboAlign = 0;
252  uint64_t ssboAlign = 0;
253 
254  // if dynamic ubo OR dynamic ssbo take the relevant alignment
255  if (static_cast<uint32_t>(usage & BufferUsageFlags::UniformBuffer) != 0 && minUboDynamicAlignment) { uboAlign = minUboDynamicAlignment; }
256  if (static_cast<uint32_t>(usage & BufferUsageFlags::StorageBuffer) != 0 && minSsboDynamicAlignment) { ssboAlign = minSsboDynamicAlignment; }
257  _minDynamicAlignment = std::max(uboAlign, ssboAlign);

265  // INTERNAL NOTE: CALL THIS ** AFTER ** CALC BASE ALIGNMENT, TO FIX THE _offset AND _size MEMBERS
266  void calcSizeAndOffset(const uint32_t& offset)
267  {
268  _offset = align(offset, _baseAlignment);
269  _offset = align(_offset, _minDynamicAlignment);
270  if (isStructure())
271  {
272  uint32_t tmp_offset = 0;
273  for (auto& child : _childEntries)
274  {
275  child.calcSizeAndOffset(tmp_offset);
276  tmp_offset = child.getOffset() + static_cast<uint32_t>(child.getSize());
277  }
278  // STD140: Structures are padded to their alignment so that a[n] ==> sizeof(a[0]) == sizeof[a] / n
279  tmp_offset = align(tmp_offset, _baseAlignment);
280  tmp_offset = align(tmp_offset, _minDynamicAlignment);
281  _arrayMemberSize = tmp_offset;
282  _singleElementSize = _arrayMemberSize;
283 
284  tmp_offset *= _numArrayElements;
285  _size = tmp_offset;
286  }
287  else
288  {
289  _arrayMemberSize = getSelfAlignedArraySize(_type);
290  _singleElementSize = pvr::getSize(_type);
291  _size = _variableArray ? _arrayMemberSize * _numArrayElements : pvr::getSize(_type, _numArrayElements);
292  }

561  const StructuredMemoryEntry& _prototype;
562  StructuredBufferViewElement(const StructuredMemoryEntry& entry, uint32_t level, uint32_t elementArrayIndex, const uint32_t* parentIndices, uint32_t dynamicSlice = 0)
563  : _mappedMemory(nullptr), _level(level), _prototype(entry)
564  {
565  _indices[0] = elementArrayIndex;
566  if (parentIndices) { memcpy(_indices + 1, parentIndices, sizeof(uint32_t) * (level)); }
567  init(dynamicSlice);
568  }
569  void init(uint32_t dynamicSlice = 0)
570  {
571  // Get the offset INSIDE CURRENT LEVEL.
572  _offset = _prototype.getArrayElementOffset(_indices[0]);
573  const StructuredMemoryEntry* parent = _prototype.getParent();
574  size_t level = 1; // How many levels up we have gone
575  uint64_t dynamicSliceSize = 0;
576  uint32_t mappedDynamicSlice = 0;
577  while (parent) // Until we reach the root element
578  {
579  debug_assertion(parent->getNumArrayElements() > _indices[level], "StructuredBufferViewElement: Attempted out-of-bounds access in getOffset");
580  _offset += parent->getArrayElementOffset(_indices[level++]);
581  dynamicSliceSize = parent->getSize();
582  mappedDynamicSlice = parent->getMappedDynamicSlice();
583  // store the mapped memory so we only have to do this lookup once rather than each time setValue is called
584  _mappedMemory = parent->getMappedMemory();
585  parent = parent->getParent();
586  }
587 
588  // at this point dynamicSliceSize matches the root size
589  debug_assertion(dynamicSlice >= mappedDynamicSlice, "StructuredBufferViewElement: Mapped dynamic slice must be greater than or equal to the current dynamic slice");
590  uint32_t sliceOffset = (dynamicSlice * static_cast<uint32_t>(dynamicSliceSize)) - (mappedDynamicSlice * static_cast<uint32_t>(dynamicSliceSize));
591  _offset += sliceOffset;
592  }

874  }

879  uint64_t myvaluesize = value.dataSize();

881  }
