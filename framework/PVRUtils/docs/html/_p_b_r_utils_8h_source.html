15 inline glm::vec2 hammersley(uint32_t i, uint32_t N)
16 {
17  // Radical inverse based on http://holger.dammertz.org/stuff/notes_HammersleyOnHemisphere.html
18  uint32_t bits = (i << 16u) | (i >> 16u);
19  bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u);
20  bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u);
21  bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u);
22  bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u);
23  float rdi = float(bits) * static_cast<float>(2.3283064365386963e-10);
24  return glm::vec2(float(i) / float(N), rdi);
25 }

36 // Sample a half-vector in world space
37 // Based on http://blog.selfshadow.com/publications/s2013-shading-course/karis/s2013_pbs_epic_slides.pdf
38 glm::vec3 importanceSampleGGX(glm::vec2 Xi, float roughness, glm::vec3 N)
39 {
40  // Maps a 2D point to a hemisphere with spread based on roughness
41  float a = roughness * roughness;
42  float phi = 2.0f * glm::pi<float>() * Xi.x;
43  float cosTheta = sqrt(glm::clamp((1.0f - Xi.y) / (1.0f + (a * a - 1.0f) * Xi.y), 0.0f, 1.0f));
44  float sinTheta = sqrt(glm::clamp(1.0f - cosTheta * cosTheta, 0.0f, 1.0f));
45 
46  glm::vec3 H = glm::vec3(sinTheta * cos(phi), sinTheta * sin(phi), cosTheta);
47 
48  glm::vec3 up = glm::abs(N.z) < 0.999 ? glm::vec3(0, 0, 1) : glm::vec3(1, 0, 0);
49  glm::vec3 tangent = glm::normalize(glm::cross(up, N));
50  glm::vec3 bitangent = glm::cross(N, tangent);
51 
52  return glm::normalize(tangent * H.x + bitangent * H.y + N * H.z);
53 }

55 // http://blog.selfshadow.com/publications/s2013-shading-course/karis/s2013_pbs_epic_notes_v2.pdf
56 glm::vec2 integrateBRDF(float roughness, float NoV)
57 {
58  const glm::vec3 N = glm::vec3(0.0, 0.0, 1.0); // normal always pointing forward.
59  const glm::vec3 V = glm::vec3(sqrt(glm::clamp(1.0 - NoV * NoV, 0.0, 1.0)), 0.0, NoV);
60  float A = 0.0f;
61  float B = 0.0f;
62 
63  uint32_t numSamples = 1024u;
64  for (uint32_t i = 0u; i < numSamples; ++i)
65  {
66  glm::vec2 Xi = hammersley(i, numSamples);
67  glm::vec3 H = importanceSampleGGX(Xi, roughness, N);
68  glm::vec3 L = 2.0f * glm::dot(V, H) * H - V;
69 
70  float NoL = glm::max(glm::dot(N, L), 0.0f);
71  if (NoL > 0.0f)
72  {
73  float NoH = glm::max(glm::dot(N, H), 0.001f);
74  float VoH = glm::max(glm::dot(V, H), 0.001f);
75  float currentNoV = glm::max(glm::dot(N, V), 0.001f);
76 
77  const float G = gSmith(NoL, currentNoV, roughness);
78 
79  const float G_Vis = (G * VoH) / (NoH * currentNoV /*avoid division by zero*/);
80  const float Fc = pow(1.0f - VoH, 5.0f);
81 
82  A += (1.0f - Fc) * G_Vis;
83  B += Fc * G_Vis;
84  }
85  }
86 
87  return glm::vec2(A, B) / float(numSamples);
88 }
