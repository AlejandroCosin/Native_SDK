44 // Enable the following defines for improved validation of memory usage in debug builds
45 // VMA_DEBUG_INITIALIZE_ALLOCATIONS:
46 // Makes memory of all new allocations initialized to bit pattern `0xDCDCDCDC`.
47 // Before an allocation is destroyed, its memory is filled with bit pattern `0xEFEFEFEF`.
48 // If you find these values while debugging your program, good chances are that you incorrectly
49 // read Vulkan memory that is allocated but not initialized, or already freed, respectively.
50 #define VMA_DEBUG_INITIALIZE_ALLOCATIONS 1
51 // VMA_DEBUG_MARGIN:
52 // Enforces a specified number of bytes as a margin before and after every allocation.
53 // If your bug goes away after enabling margins, it means it may be caused by memory
54 // being overwritten outside of allocation boundaries.It is not100 % certain though. Change in application behavior may also be caused by different order and
55 // distribution of allocations across memory blocks after margins are applied.
56 #define VMA_DEBUG_MARGIN 4
57 // VMA_DEBUG_DETECT_CORRUPTION:
58 // If this feature is enabled, number of bytes specified as `VMA_DEBUG_MARGIN`
59 // (it must be multiply of 4) before and after every allocation is filled with a magic number. This idea is also know as "canary".
60 // Memory is automatically mapped and unmapped if necessary. This number is validated automatically when the allocation is destroyed.
61 // If it's not equal to the expected value, `VMA_ASSERT()` is executed. It clearly means that either CPU or GPU overwritten the
62 // memory outside of boundaries of the allocation, which indicates a serious bug.
63 #define VMA_DEBUG_DETECT_CORRUPTION 1
64 #endif

973  if (static_cast<uint32_t>(_flags & pvrvk::MemoryPropertyFlags::e_HOST_COHERENT_BIT) != 0)
974  {
975  Log(LogLevel::Warning,
976  "Invalidating range of an allocation 0x%llx from memory block 0x%llx"
977  " created using HOST_COHERENT_BIT memory flags - this is unnecessary.",
978  _vmaAllocation, getVkHandle());
979  }
980 
981  VkMappedMemoryRange range = {};
982  range.sType = static_cast<VkStructureType>(pvrvk::StructureType::e_MAPPED_MEMORY_RANGE);
983  range.memory = getVkHandle();
984  range.offset = offset;
985  range.size = size;
986  pvrvk::impl::vkThrowIfFailed(_device.lock()->getVkBindings().vkInvalidateMappedMemoryRanges(_device.lock()->getVkHandle(), 1, &range), "Failed to invalidate range of memory block");
987 }

990 {
991  offset += getOffset();
992  if (size == VK_WHOLE_SIZE) { size = getOffset() + getSize() - offset; }
993  assert(size <= _allocInfo.size);
994 }
