206  const GLuint& getFontIbo()
207  {
208  if (!_fontIboCreated)
209  {
210  // create the FontIBO
211  std::vector<uint16_t> fontFaces;
212  fontFaces.resize(impl::Font_::FontElement);
213 
214  for (uint32_t i = 0; i < impl::Font_::MaxRenderableLetters; ++i)
215  {
216  fontFaces[i * 6u] = static_cast<uint16_t>(0u + i * 4u);
217  fontFaces[i * 6u + 1u] = static_cast<uint16_t>(3u + i * 4u);
218  fontFaces[i * 6u + 2u] = static_cast<uint16_t>(1u + i * 4u);
219 
220  fontFaces[i * 6u + 3u] = static_cast<uint16_t>(3u + i * 4u);
221  fontFaces[i * 6u + 4u] = static_cast<uint16_t>(0u + i * 4u);
222  fontFaces[i * 6u + 5u] = static_cast<uint16_t>(2u + i * 4u);
223  }
224  GLint binding;
225  gl::GetIntegerv(GL_ELEMENT_ARRAY_BUFFER_BINDING, &binding);
226 
227  gl::GenBuffers(1, &_fontIbo);
228  gl::BindBuffer(GL_ELEMENT_ARRAY_BUFFER, _fontIbo);
229  gl::BufferData(GL_ELEMENT_ARRAY_BUFFER, static_cast<uint32_t>(sizeof(fontFaces[0]) * fontFaces.size()), &fontFaces[0], GL_STATIC_DRAW);
230  gl::BindBuffer(GL_ELEMENT_ARRAY_BUFFER, binding);
231 
232  _uiStateTracker.ibo = _fontIbo;
233  _fontIboCreated = true;
234  }
235  return _fontIbo;
236  }
