80

84  frustum_out.plusY = row3 - row1;
85  frustum_out.minusZ = row3 + row2;
86  frustum_out.plusZ = row3 - row2;

101 inline glm::vec3 IntersectPlanes(glm::vec4 p0, glm::vec4 p1, glm::vec4 p2)
102 {
103  glm::vec3 bxc = glm::cross(glm::vec3(p1), glm::vec3(p2));
104  glm::vec3 cxa = glm::cross(glm::vec3(p2), glm::vec3(p0));
105  glm::vec3 axb = glm::cross(glm::vec3(p0), glm::vec3(p1));
106  glm::vec3 r = -p0.w * bxc - p1.w * cxa - p2.w * axb;
107  return r * (1 / glm::dot(glm::vec3(p0), bxc));
108 }
