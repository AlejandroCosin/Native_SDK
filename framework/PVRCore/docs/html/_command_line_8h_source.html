410  void parseCmdLine(const char* const cmdLine)
411  {
412  size_t len;
413  uint32_t nIn, nOut;
414  bool bInQuotes;
415  ParsedCommandLine::Option opt;
416 
417  if (cmdLine == nullptr) { return; }
418 
419  // Take a copy of the original
420  len = strlen(cmdLine) + 1;
421 
422  // Take a copy to be edited
423  _data.resize(len);
424 
425  // Break the command line into options
426  bInQuotes = false;
427  opt.arg = nullptr;
428  opt.val = nullptr;
429  nIn = static_cast<uint32_t>(-1);
430  nOut = 0;
431  do
432  {
433  ++nIn;
434  if (cmdLine[nIn] == '"') { bInQuotes = !bInQuotes; }
435  else
436  {
437  if (bInQuotes && cmdLine[nIn] != 0)
438  {
439  if (opt.arg == nullptr) { opt.arg = &_data[nOut]; }
440 
441  _data[nOut++] = cmdLine[nIn];
442  }
443  else
444  {
445  switch (cmdLine[nIn])
446  {
447  case '=':
448  _data[nOut++] = 0;
449  opt.val = &_data[nOut];
450  break;
451 
452  case ' ':
453  case '\t':
454  case '\0':
455  _data[nOut++] = 0;
456  if ((opt.arg != nullptr) || (opt.val != nullptr))
457  {
458  // Add option to list
459  _commandLine._options.emplace_back(opt);
460 
461  opt.arg = nullptr;
462  opt.val = nullptr;
463  }
464  break;
465 
466  default:
467  if (opt.arg == nullptr) { opt.arg = &_data[nOut]; }
468 
469  _data[nOut++] = cmdLine[nIn];
470  break;
471  }
472  }
473  }
474  } while (cmdLine[nIn] != 0);
