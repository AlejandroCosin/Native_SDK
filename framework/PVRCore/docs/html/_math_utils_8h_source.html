227 inline void addCoefficients(const uint64_t pascalSum, const size_t halfCoefficientsMinusOne, const size_t numCoefficients, const std::vector<uint64_t>& coefficients,
228  std::vector<double>& weights, std::vector<double>& offsets)
229 {
230  // Handle cases where the coefficients vector contains coefficients which are to be truncated. i.e. we get rid of the outer set of coefficients
231  size_t unneededCoefficients = (coefficients.size() - static_cast<size_t>(numCoefficients)) / 2;
232  size_t i = unneededCoefficients;
233  for (; i < (size_t)halfCoefficientsMinusOne; i++)
234  {
235  weights.emplace_back(static_cast<double>(coefficients[i]) / pascalSum);
236  double offset = static_cast<double>(i) - static_cast<double>(halfCoefficientsMinusOne);
237  offsets.emplace_back(offset);
238  }
239 
240  weights.emplace_back(static_cast<double>(coefficients[i]) / pascalSum);
241  offsets.emplace_back(static_cast<double>(0));
242 
243  for (i = static_cast<size_t>(halfCoefficientsMinusOne) + 1; i < static_cast<size_t>(numCoefficients) + unneededCoefficients; i++)
244  {
245  weights.emplace_back(static_cast<double>(coefficients[i]) / pascalSum);
246  offsets.emplace_back(static_cast<double>(i - halfCoefficientsMinusOne));
247  }
248 }

259 inline uint64_t generatePascalTriangleRow(size_t row, std::vector<uint64_t>& pascalCoefficients)
260 {
261  // Each entry of any given row of the Pascal Triangle is constructed by adding the number above and to the left with the number above and to the right.
262  // Entries which fall outside of the Pascal Triangle are treated as having a 0 value. The first row consists of a single entry with a value of 1.
263  // The following shows the first 4 rows of the Pascal Triangle
264  // 1 row 0 ... sum = 1
265  // 1 1 row 1 ... sum = 2
266  // 1 2 1 row 2 ... sum = 4
267  // 1 3 3 1 row 3 ... sum = 8
268  pascalCoefficients.emplace_back(1u);
269  uint64_t sum = pascalCoefficients.back();
270  for (size_t i = 0; i < row; i++)
271  {
272  uint64_t val = pascalCoefficients[i] * (row - i) / (i + 1u);
273  pascalCoefficients.emplace_back(val);
274  sum += pascalCoefficients.back();
275  }
276 
277  return sum;
278 }

285 inline void adjustOffsetsAndWeightsForLinearSampling(const size_t halfCoefficientsMinusOne, std::vector<double>& weights, std::vector<double>& offsets)
286 {
287  std::vector<double> adjustedWeights;
288  std::vector<double> adjustedOffsets;
289 
290  // if kernel size minus 1 is divisible by 2 then we have a central sample with offset 0
291  if (halfCoefficientsMinusOne % 2u == 0u)
292  {
293  size_t i = 0u;
294  // Ensure not zero
295  for (; (halfCoefficientsMinusOne > 0) && (i < halfCoefficientsMinusOne - 1u); i += 2u)
296  {
297  adjustedWeights.emplace_back(weights[i] + weights[i + 1]);
298  double adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1u] * weights[i + 1u])) / adjustedWeights.back();
299  adjustedOffsets.emplace_back(adjustedOffset);
300  }
301 
302  adjustedWeights.emplace_back(weights[halfCoefficientsMinusOne]);
303  adjustedOffsets.emplace_back(0.0f);
304 
305  for (i = halfCoefficientsMinusOne + 1u; i < static_cast<uint64_t>(offsets.size()); i += 2u)
306  {
307  adjustedWeights.emplace_back(weights[i] + weights[i + 1u]);
308  double adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1u] * weights[i + 1u])) / adjustedWeights.back();
309  adjustedOffsets.emplace_back(adjustedOffset);
310  }
311  }
312  else // otherwise we have to duplicate the central sample *but* this means we can handle 3x3 using 2x2 samples
313  {
314  size_t i = 0;
315  for (; i < halfCoefficientsMinusOne; i += 2u)
316  {
317  double adjustedOffset = 0.0;
318  if (i == halfCoefficientsMinusOne - 1u)
319  {
320  adjustedWeights.emplace_back(weights[i] + weights[i + 1u] * 0.5);
321  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1u] * weights[i + 1u] * 0.5)) / adjustedWeights.back();
322  }
323  else
324  {
325  adjustedWeights.emplace_back(weights[i] + weights[i + 1u]);
326  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1u] * weights[i + 1u])) / adjustedWeights.back();
327  }
328  adjustedOffsets.emplace_back(adjustedOffset);
329  }
330 
331  for (i = halfCoefficientsMinusOne; i < offsets.size(); i += 2u)
332  {
333  double adjustedOffset = 0.0;
334  if (i == halfCoefficientsMinusOne)
335  {
336  adjustedWeights.emplace_back(weights[i] * 0.5 + weights[i + 1u]);
337  adjustedOffset = ((offsets[i] * weights[i] * 0.5) + (offsets[i + 1u] * weights[i + 1u])) / adjustedWeights.back();
338  }
339  else
340  {
341  adjustedWeights.emplace_back(weights[i] + weights[i + 1u]);
342  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1u] * weights[i + 1u])) / adjustedWeights.back();
343  }
344  adjustedOffsets.emplace_back(adjustedOffset);
345  }
346  }
347 
348  offsets.clear();
349  weights.clear();
350 
351  for (size_t i = 0; i < adjustedWeights.size(); i++)
352  {
353  offsets.emplace_back(adjustedOffsets[i]);
354  weights.emplace_back(adjustedWeights[i]);
355  }
356 }

376 inline void generateGaussianKernelWeightsAndOffsets(uint32_t kernelSize, bool truncateCoefficients, bool useLinearSamplerOptimization, std::vector<double>& weights,
377  std::vector<double>& offsets, float minimumAcceptableCoefficient = 0.0001f)
378 {
379  // Odd kernel sizes are a requirement
380  assert((kernelSize % 2u) == 1u);
381 
382  // The starting row of the Pascal Triangle being used
383  size_t pascalRow = kernelSize - 1u;
384 
385  // The number of coefficients minus 1 halved
386  // This variable is used to get the index of the coefficient used for an offset of 0
387  // (numCoefficients - 1) / 2
388  size_t halfCoefficientsMinusOne = pascalRow / 2u;
389 
390  // stores the set of pascal coefficients
391  std::vector<uint64_t> pascalCoefficients;
392 
393  // the number of coefficients skipped due to minimum coefficient checks
394  size_t numCoefficientsSkipped = 0u;
395 
396  // The simple case
397  if (!truncateCoefficients)
398  {
399  // Generate the Pascal Triangle row and calculate the sum of the row
400  uint64_t pascalSum = generatePascalTriangleRow(pascalRow, pascalCoefficients);
401  // Store the Pascal Triangle coefficients for the given row
402  addCoefficients(pascalSum, halfCoefficientsMinusOne, pascalCoefficients.size(), pascalCoefficients, weights, offsets);
403  }
404  else
405  // Ignoring negligible coefficients - we'll now attempt to find a row which provides enough coefficients for the given kernel size whilst not falling below the
406  // given minimal coefficient value.
407  {
408  size_t currentRow = pascalRow;
409  bool foundSuitableRow = false;
410 
411  // only accept rows where we have kernelSize coefficients larger than the min coefficient
412  while (!foundSuitableRow)
413  {
414  // clear the offsets and weights
415  offsets.clear();
416  weights.clear();
417 
418  // clear the set of pascal triangle coefficients
419  pascalCoefficients.clear();
420 
421  // get the pascal coefficients for the current row of the triangle
422  uint64_t pascalSum = generatePascalTriangleRow(currentRow, pascalCoefficients);
423 
424  halfCoefficientsMinusOne = currentRow / 2u;
425 
426  // check how many of the coefficients are negligible in size and therefore should be ignored
427  numCoefficientsSkipped = 0u;
428  for (size_t i = halfCoefficientsMinusOne; i < pascalCoefficients.size(); i++)
429  {
430  double currentWeight = static_cast<double>(pascalCoefficients[i]) / pascalSum;
431  if (currentWeight < minimumAcceptableCoefficient) { numCoefficientsSkipped++; }
432  }
433 
434  // if there aren't enough coefficients left to fulfill the requirements for the kernel size then continue to the next row
435  if ((halfCoefficientsMinusOne + 1u) - numCoefficientsSkipped < (pascalRow / 2u + 1u))
436  {
437  currentRow += 2u;
438  foundSuitableRow = false;
439  continue;
440  }
441  // if negligible coefficients are to be removed we must also update the overall sums used to give them weighting
442  // otherwise repeated blurring will result in darkening of the image
443  uint64_t adjustedPascalSum = pascalSum;
444  // the extra coefficients which aren't needed which do match the conditions for non-negligibility but would result in us taking extra coefficients
445  size_t unrequiredCoefficients = ((pascalCoefficients.size() - kernelSize - (numCoefficientsSkipped * 2u)) / 2u);
446  for (size_t i = 0; i < numCoefficientsSkipped + unrequiredCoefficients; i++) { adjustedPascalSum -= 2u * pascalCoefficients[pascalCoefficients.size() - 1u - i]; }
447 
448  // add the non-negligible coefficients to the weights and offsets buffers
449  size_t numCoefficients = pascalCoefficients.size() - (2u * (numCoefficientsSkipped + unrequiredCoefficients));
450  addCoefficients(adjustedPascalSum, halfCoefficientsMinusOne, numCoefficients, pascalCoefficients, weights, offsets);
451  halfCoefficientsMinusOne = static_cast<uint64_t>((offsets.size() - 1u) / 2u);
452  foundSuitableRow = true;
453  }
454  }
455 
456  // If using the Linear Sampling optimisation then adjust the Gaussian weights and offsets accordingly
457  if (useLinearSamplerOptimization) { adjustOffsetsAndWeightsForLinearSampling(halfCoefficientsMinusOne, weights, offsets); }
458 }
