348  void compact()
349  {
350  // We can do that because the last remove() tears down all datastructures used.
351  if (!myindex.size()) { return; }
352  // First, make sure there is something to compact...
353  deleteditemlisttype_::iterator unused_spot = myDeletedItems.begin();
354  while (myDeletedItems.size() && unused_spot != myDeletedItems.end())
355  {
356  // Last item in the storage vector.
357  size_t last = mystorage.size();
358 
359  // 1) Trim the end of the vector...
360  while (last-- && mystorage[last].isUnused)
361  {
362  mystorage.pop_back();
363  // Rinse, repeat. If size is zero, there is nothing to do.
364  }
365 
366  // Either the storage is empty
367  if (mystorage.empty())
368  {
369  myDeletedItems.clear();
370  // the rest should already be cleared...
371  // the loop will exit naturally
372  }
373  else // Or we can have find its last item!
374  {
375  // 2)Trim any items from the end of the unused spots list that may have been trimmed off by the vector...
376  while (unused_spot != myDeletedItems.end() && *unused_spot >= last) { myDeletedItems.erase(unused_spot++); }
377  // Any spots left?
378  if (unused_spot != myDeletedItems.end())
379  {
380  // Do the actual data movement. After all we've been through, we know that
381  // i. The last item of the vector is a valid item (guaranteed by 1)
382  // ii. The unused spot is not out of bounds of the vector
383  // iii.
384  // Copy by hand(as we have not defined a move assignment operator for compatibility reasons.
385  // Also, since the std::string does not throw, this makes easier to reason about exceptions.
386 
387  mystorage[*unused_spot].value = std::move(mystorage[last].value);
388  mystorage[*unused_spot].key = std::move(mystorage[last].key);
389 
390  mystorage[*unused_spot].isUnused = false;
391  myindex[mystorage[*unused_spot].key] = *unused_spot;
392  mystorage.pop_back();
393  myDeletedItems.erase(unused_spot++);
394  } // else : No action needed - unused spots has been trimmed off completely, so no movement is possible, or necessary...
395  }
396  }

428  bool relocate(const IndexType_& key, size_t index)
429  {
430  typename maptype_::iterator found = myindex.find(key);
431  if (found == myindex.end()) { return false; }
432  size_t old_index = myindex[key];
433  if (index == old_index) { return true; } // No-op
434  if (index + 1 > mystorage.size()) // Storage not big enough.
435  {
436  // Grow, and mark unused all required items. Need to add all spots (But the last) to unusedspots.
437  size_t oldsize = mystorage.size();
438  mystorage.resize(index + 1);
439  for (size_t i = oldsize; i < index; ++i)
440  {
441  myDeletedItems.push_front(i);
442  mystorage[i].isUnused = 1;
443  }
444  mystorage.back() = mystorage[old_index];
445  removefromvector(old_index);
446  }
447  else if (mystorage[index].isUnused) // Lucky! Storage is big enough, and the item is not used!
448  {
449  deleteditemlisttype_::iterator place = std::find(myDeletedItems.begin(), myDeletedItems.end(), index);
450  assert(place != myDeletedItems.end()); // Shouldn't happen! Ever!
451  myDeletedItems.erase(place);
452  mystorage[index] = mystorage[old_index];
453  removefromvector(old_index);
454  }
455  else // Whoops! Space is already occupied. Swap with the old item!
456  {
457  myindex[mystorage[index].key] = old_index;
458  std::swap(mystorage[index], mystorage[old_index]);
459  }
460  myindex[key] = index;
461  return true;
462  }
